function getApiKey() {
  return localStorage.getItem('GEMINI_API_KEY') || '';
}

// Error classifier
function classifyError(status, body) {
  const txt = JSON.stringify(body || '');
  if (status === 400 && (txt.includes('API_KEY') || txt.includes('api key') || txt.includes('INVALID_ARGUMENT')))
    return 'INVALID_KEY';
  if (status === 429 || txt.includes('RESOURCE_EXHAUSTED') || txt.includes('quota'))
    return 'RATE_LIMIT';
  if (status === 403) return 'FORBIDDEN';
  return 'ERROR';
}

// Core Gemini call — with optional Google Search grounding
async function callGemini({ model = MODEL_FLASH, systemPrompt, userPrompt, useSearch = true }) {
  const apiKey = getApiKey();
  if (!apiKey) throw { type: 'NO_KEY', message: 'Gemini API key not set. Please enter your key in Settings.' };

  const url = `https://generativelanguage.googleapis.com/v1beta/models/${model}:generateContent?key=${apiKey}`;

  const body = {
    systemInstruction: { parts: [{ text: systemPrompt }] },
    contents: [{ role: "user", parts: [{ text: userPrompt }] }],
    generationConfig: { temperature: 0.2, maxOutputTokens: 2048 },
  };

  // Add Google Search grounding (same as your AI Studio googleSearch tool)
  if (useSearch) {
    body.tools = [{ googleSearch: {} }];
  }

  const res = await fetch(url, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify(body),
  });

  const data = await res.json();

  if (!res.ok) {
    const errType = classifyError(res.status, data);
    const errMsg = data?.error?.message || `HTTP ${res.status}`;
    throw { type: errType, message: errMsg, status: res.status };
  }

  // Extract text from response
  const text = data?.candidates?.[0]?.content?.parts
    ?.filter(p => p.text)
    ?.map(p => p.text)
    ?.join('') || '';

  // Check for safety block
  const finishReason = data?.candidates?.[0]?.finishReason;
  if (finishReason === 'SAFETY' || finishReason === 'RECITATION') {
    throw { type: 'BLOCKED', message: 'Response was blocked by safety filters. Try rephrasing.' };
  }

  return text;
}

// Smart caller: tries with Google Search, falls back without search on rate limit
async function smartGemini(systemPrompt, searchPrompt, fallbackPrompt) {
  try {
    const text = await callGemini({ systemPrompt, userPrompt: searchPrompt, useSearch: true });
    return { text, mode: 'search' };
  } catch (e) {
    if (e?.type === 'NO_KEY' || e?.type === 'INVALID_KEY' || e?.type === 'FORBIDDEN') throw e;
    if (e?.type === 'RATE_LIMIT' || e?.status >= 500) {
      // Fallback: no search grounding
      const text = await callGemini({ systemPrompt, userPrompt: fallbackPrompt, useSearch: false });
      return { text, mode: 'ai' };
    }
    throw e;
  }
}
